SECTION 03 Window Function — OVER (PARTITION BY)
Question: Show each employee with their department average — keeping all 7 rows intact.
Window AVG
SELECT EmployeeName, Department, Salary,
    AVG(Salary) OVER (PARTITION BY Department)
        AS DepartmentAverageSalary
FROM Employees;   -- ALL 7 rows returned, with dept avg attached
Window SUM
SELECT EmployeeName, Department, Salary,
    SUM(Salary) OVER (PARTITION BY Department)
        AS TotalSalaryPerDept
FROM Employees;   -- ALL 7 rows returned, with dept total attached
Window COUNT
SELECT EmployeeName, Department,
    COUNT(*) OVER (PARTITION BY Department)
        AS EmployeesInDept
FROM Employees;   -- ALL 7 rows returned, with headcount attached
